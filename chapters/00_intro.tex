\section{Setting the stage}

\begin{frame}{Where we are in this talk}
  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
      \textbf{Goal of the seminar.}\\
      Take a custom mobile platform and let \textbf{Nav2} drive it autonomously.
      \vspace{1.0em}

      \textbf{Today, two halves:}
      \begin{enumerate}
        \item \textbf{Part~I — Gap analysis}\\
              \footnotesize\textcolor{black!60}{What does Nav2 actually deliver,
              and what is still missing before a wheel can turn?}
        \item \textbf{Part~II — The recipe}\\
              \footnotesize\textcolor{black!60}{ros2\_control, custom hardware
              plugin, CAN contract, firmware on the MCU.}
      \end{enumerate}

      \vspace{0.4em}
      \footnotesize\textcolor{black!60}{All on the \textbf{MiviaRover}, our
      outdoor UGV — case study, not subject.}

    \column{0.45\textwidth}
      \begin{tikzpicture}[node distance=4mm and 6mm]
        \node[hl]                    (nav2)  {Nav2 stack};
        \node[ghost, below=of nav2]  (gap)   {\textbf{?}\\\scriptsize the gap};
        \node[ll,    below=of gap]   (mcu)   {Motors\\(MCU)};
        \draw[arrhl] (nav2) -- (gap);
        \draw[arrgap] (gap) -- (mcu);
      \end{tikzpicture}
  \end{columns}
\end{frame}

\begin{frame}{Why this talk}
  \begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
    \textbf{The thesis.}\;
Nav2 is well-documented \emph{upstream} of \texttt{cmd\_vel}.
    \emph{Downstream}: a \alert{grey area} no tutorial covers end-to-end.
\vspace{0.7em}

    \textbf{The promise.}\;
    By the end you will understand \emph{exactly} what belongs in that grey area — and be able to build your own.

\vspace{0.7em}
      \textbf{Method.}\;
      Each concept: \emph{principle} $\rightarrow$
      \emph{MiviaRover instance}. Pseudocode and block diagrams.
    \column{0.40\textwidth}
      \footnotesize
      \textcolor{black!60}{Out of scope}
      \begin{itemize}\setlength\itemsep{3pt}
        \item \emph{Localization} (EKF + UWB) — next talk, E.~Amato.
        \item \emph{Perception} (ZED~2i) — assumed available.
      \end{itemize}
      \vspace{0.6em}
      \textcolor{black!60}{Audience}
      \begin{itemize}\setlength\itemsep{3pt}
        \item \emph{Intelligent Robot Navigation} students.
        \item ROS~2 basics assumed; Q\&A at the end.
      \end{itemize}
  \end{columns}
\end{frame}
