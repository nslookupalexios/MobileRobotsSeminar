\section{Nav2 in one slide}

\begin{frame}{Recap: the Navigation~2 server pipeline}
  \centering
  \includegraphics[width=\linewidth,height=0.75\textheight,keepaspectratio]{images/02_nav2_stack.png}

  \vspace{0.5em}
  \begin{itemize}\setlength\itemsep{2pt}
    \item Each block is a \textbf{lifecycle node}; the pipeline is orchestrated by a Behaviour Tree.
    \item The \alert{output of the chain} is a single message: a \textbf{velocity command}.
    \item Everything to the right of the Velocity Smoother is \emph{your} responsibility.
  \end{itemize}
\end{frame}

% ---------------------------------------------------------------------------- %
\begin{frame}[fragile]{The output boundary: \texttt{cmd\_vel}}
  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
      Nav2 outputs \textbf{one topic}, body frame, at \(\sim\)20–50~Hz:

      \vspace{0.5em}
      \centering
      \fbox{\parbox{0.88\linewidth}{\centering
        \texttt{/cmd\_vel}\\
        \texttt{geometry\_msgs/TwistStamped}\\
        \scriptsize frame: \texttt{base\_link}
      }}

    \column{0.45\textwidth}
      \begin{lstlisting}[language=,basicstyle=\ttfamily\small]
Vector3 linear   # vx vy vz
Vector3 angular  # wx wy wz
      \end{lstlisting}

      \vspace{0.4em}
      Diff-drive: only two fields matter:
      \begin{itemize}
        \item \(v_x\) — forward speed \([m/s]\)
        \item \(\omega_z\) — yaw rate \([rad/s]\)
      \end{itemize}
  \end{columns}
\end{frame}
